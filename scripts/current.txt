template Position
    position
    type
end

template Sensor
    sensorId
    type
end

####

Position moldingPallet
    type = "pallet"
    position = "moldingArea_palletPlace"
end

Position warehouse_pos1
    type = "pallet"
    position = "warehouse_destination_pos"
end

Sensor buttonPalletIsReady
    sensorId = "buttonPalletIsReady"
    type = "Boolean"
end

task Refill
    Transport
    from warehouse_pos1
    to moldingPallet 
end

#### 
task Transport_moldingPallet
    Transport
    from moldingPallet
    to warehouse_pos1
    TriggeredBy	buttonPalletIsReady.value <= 6
    OnDone Refill
end
